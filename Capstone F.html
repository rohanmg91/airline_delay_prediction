def convert(flights, airport):
    
    """ Function to return DataFrame with origin airport and destination airport as enterd by the user."""
    
    origin = raw_input('Kindly enter the origin City: ').title()   
    #print (origin)
    
    if origin in airport.CITY.values:
        ind = airport[airport.CITY == origin].index[0]
        originCode = airport.IATA_CODE[ind]
        #print (originCode)
    else:
        originCode = raw_input('Please type right name of Origin Airport: ')
    
    destination = raw_input('Kindly enter the destination City: ').title()
    #print (destination)
        
    if destination in airport.CITY.values:
        ind1 = airport[airport.CITY == destination].index[0]
        destinationCode = airport.IATA_CODE[ind1]
        #print (destinationCode)
        #print type(destinationCode)
    else:
        destinationCode = raw_input('Please type right name of Destination Airport: ')
        
    newDataFrame = flights[(flights.ORIGIN_AIRPORT == originCode) & (flights.DESTINATION_AIRPORT == destinationCode)]
    return newDataFrame

def metric():
